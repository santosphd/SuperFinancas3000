CREATE DATABASE gestao_financeira

CREATE TABLE tipo_despesas(
	codigo INTEGER NOT NULL AUTO_INCREMENT,
	descricao VARCHAR (50) NOT NULL,
	plano_de_contas VARCHAR (50) NOT NULL,
	data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (codigo));

CREATE TABLE forma_pagamento(
	codigo INTEGER NOT NULL AUTO_INCREMENT,
	descricao VARCHAR (30) NOT NULL,
	banco VARCHAR (30), 
	agencia VARCHAR (6), 
	conta VARCHAR (15), 
	bandeira VARCHAR (15),
	data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (codigo));

CREATE TABLE despesas(
	id INTEGER NOT NULL AUTO_INCREMENT,
	documento VARCHAR (15) NOT NULL,
	data_despesa DATE,
	observacoes VARCHAR (50),
	valor FLOAT,
	condicao_pagamento VARCHAR (10),
	gerar_parcelas INTEGER,
	data_pagamento DATE,
	data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	codigo_tipo_despesas INTEGER,
	codigo_forma_pagamento INTEGER,
	PRIMARY KEY (id));

CREATE TABLE contas_pagar(
	id INTEGER NOT NULL AUTO_INCREMENT,
	id_despesas INTEGER,
	parcela INTEGER,
	data_cadastro TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	PRIMARY KEY (id));
